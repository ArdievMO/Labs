#include <iostream>
#include <list>

using namespace std;

void print_list(list<int>spisok)
{
	int size_list = spisok.size();
	if (size_list == 0)
	{
		cout << "Пустой список\n";
	}
	else
	{
		for (int i = 0; i < size_list; i++)
		{
			cout << spisok.front();
			spisok.pop_front();
		}
		cout << endl;
	}
}
