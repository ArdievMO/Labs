#include <iostream>
#include <queue>

using namespace std;

void print(queue<char> st)
{
	for (int i = 0; i < st.size(); i++)
	{
		cout << st.front() << "\t";
		st.push(st.front());
		st.pop();
	}
	cout << endl;
};

{

	cout << "Введите размер очереди\n";

	while (size < 1)
	{
		cout << "Ошибка! Введите корректный размер очереди\n";
		cin >> size;
	}
	queue<char> st;	
	cout << "Введите элементы очереди\n";
	for (int i = 0; i < size; i++)
	{
		char input;
		cin >> input;
		st.push(input);
	}
	cout << "ОЧЕРЕДЬ:\n";
	print(st);
	bool flag = false;
	cout << "Введите ключ для информационного поля, после которого нужно добавить элемент\n";
	char key;	
	cin >> key;
	int n = st.size();
	for (int i = 0; i < n; i++)
	{
		if (st.front() == key)

			flag = true;
			st.push(st.front());

			char add;

			st.push(st.front());
