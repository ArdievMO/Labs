#include <iostream>

using namespace std;

struct Stack
{
	int data;
	Stack* poss;
};

Stack* create_stack(int n)
{
	if (n == 0)
		return 0;
	Stack* top = 0, * stek = new Stack;
	int input, i = 2;
	cin >> input;
	stek->data = input;
	stek->poss = 0;
	top = stek;
	while (i <= n)
	{
		Stack* stek_1 = new Stack;
		cin >> input;
		stek_1->data = input;
		stek_1->poss = top;
		top = stek_1;
		i++;
	}
	return top;
};

void print(Stack* top)
{
	if (top == 0)
	{
		cout << "Пустой стек!\a" << endl;
	}
	else
	{
		Stack* stek = top;
		while (stek != 0)
		{
			cout << stek->data << "\t";
			stek = stek->poss;
		}
		cout << endl;
	}
};

int pop(Stack*& top)
{
	Stack* stek = top;
	int k = 0;
	while (stek != NULL)
	{
		k++;
		stek = stek->poss;
	}
	stek = top;
	if (k == 1)
	{
		int data_point = stek->data;
		delete stek;
		top = 0;
		return data_point;
	}
	else
	{
		Stack* pointer = stek->poss;
		int data_point = stek->data;
		top = pointer;
		delete stek;
		return data_point;
	}
};

Stack* push(Stack*& top, int data_in)
{
	Stack* stek = new Stack;
	stek->data = data_in;
	stek->poss = top;
	top = stek;
	return top;
};

int score(Stack* top)
{
	int size = 0;
	Stack* st = top;
	while (st != NULL)
	{
		st = st->poss;
		size++;
	}
	return size;
}

	Stack* structure = create_stack(size);
	cout << "Стек: \n";

	Stack* structure_2 = create_stack(0);

	for (i = 0; i < size; i++)
	{
		if (structure->data % 2 == 0)
		{
			key = size - i - 1;
		}
		push(structure_2, pop(structure));
	}

	for (i = 0; i < size; i++)
	{
		if (i == key)
		{
			pop(structure_2);
		}
		else
			push(structure, pop(structure_2));

	}
	int new_size = score(structure);
	if (new_size == size)

	else
